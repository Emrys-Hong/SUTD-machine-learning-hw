'''
Learning process
- input : x -> intensity and symmetry values, y -> true class labels
- stopping criteria : stops if all samples are correctly classified or max epoch is reached
- output : list of parameters updated at each mistake
'''
def perceptron_train (data_set,epoch) :
    
    x = data_set[:,:-1]
    y = data_set[:,-1]
    
    #initialize parameters (both weight vector elements and bias term is set to 0)
    theta = np.zeros(len(x[0]))
    theta_note = 0
    num_epoch = epoch
    theta_updated = [theta]
    theta_note_updated = [theta_note]
    
    for t in range(num_epoch): # t starts from 0 not 1
        for i in range(0, len(x)):
            x[i] = np.array(x[i])
            y[i] = np.array(y[i])
            
            # if a mistake was found (sign of the predicted value != sign of true label)
            if y[i]*(np.dot(theta_updated[-1], x[i]) + theta_note_updated[-1]) <= 0 :
                theta_new = theta_updated[-1] + y[i]*np.array(x[i])
                theta_note_new = theta_note_updated[-1] + y[i]
                # add updated values to param list
                theta_updated.append(theta_new)
                theta_note_updated.append(theta_note_new)
                
        # Calculate 0-1 loss using final updated parameters for this iteration
        training_error = 0
        for i in range(0, len(x)):
            training_error += 1 if y[i]*(np.dot(theta_updated[-1], x[i]) + theta_note_updated[-1]) <= 0 else 0
            
        if training_error == 0:
            print("0-1 loss is 0 in iteration ", t+1, ' after ', len(theta_updated)-1, 'parameter updates')
            break
        
        if t == num_epoch-1:
            print('Average 0-1 loss = ' + str(float(training_error)/len(x)))
    
    return theta_updated, theta_note_updated

'''
Evaluating process
- input : x -> intensity and symmetry values, y -> true labels, 
        theta and theta_note -> optimal paramters obtained from traning process
- use sign function to assign label to data based on score of x.theta + theta_note
- output : performance values based on metrics -> accuracy
'''       
def perceptron_eval (x, y, theta, theta_note) :
    
    predicted_labels = [1 if (np.dot(theta, doc) + theta_note) > 0 else -1 for doc in x]
    
    accuracy = accuracy_score(y, predicted_labels)
    
    return accuracy

def cal_cost(theta, X, y):
    '''
    theta: weights, vector
    X: feautes, array
    y: labels, vector
    '''
    m = len(y)
    prediction = X.dot(theta)
    cost = (1/(2.0*m))*np.sum(np.square(prediction-y))
    return cost

def gradient_descent(X,y,theta,learning_rate,iterations):
    '''
    X    = Matrix of X with added bias units
    y    = Vector of Y
    theta=Vector of thetas np.random.randn(j,1)
    learning_rate 
    iterations = no of iterations
    
    Returns the final theta vector and array of cost history over no of iterations
    '''

    m = len(y)
    cost_history = np.zeros(iterations)
    theta_history = np.zeros((iterations, 2))
    
    for it in range(iterations):
        cost = 0.0
        for i in range(m):
            X_i = X[i,:].reshape(1,X.shape[1])
            y_i = y[i].reshape(1,1)
            prediction = np.dot(X_i,theta)

            theta = theta + learning_rate*(X_i.T.dot((y_i-prediction)))
            cost += cal_cost(theta,X_i,y_i) 
        cost_history[it] = cost
        theta_history[it,:] = theta.T
        #print theta.T, cost

    return theta_history, cost_history

def PolyRegress(X,y,d=2):
    '''
    X: features (without bias), vector
    y: labels, vector
    d: order, greater than 0
    '''
    assert d>0
    X_poly = []
    X = np.array(X)
    #high order features
    for i in range(d+1):
        X_poly.append(X**i)
    X_poly = np.array(X_poly)
    #closed form regression
    theta = closed_form_regress(X_poly.T, y)
    #calculate the cost
    cost = cal_cost(theta, X_poly.T, y)
    return theta, X_poly, cost

def ridge_regression(tX, tY, l=0.15):
  '''
  tX: features, array
  tY: labels, vector
  l: regularization term, float
  '''
  n, dim = tX.shape
  diag = np.array([n*l]*dim)
  A = np.diag(diag)+ np.dot(tX.T, tX)
  b = np.dot(tX.T, tY)
  theta = np.dot(np.linalg.inv(A), b)
  return theta
